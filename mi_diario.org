*** Grabar imagen en pendrive o sd
#+BEGIN_SRC sh
    # localizar la imagen y descomprimirla
    unzip loquesea.zip
    # localizar la unidad del pendrive
    df -h
    # desmontar la unidad (o varias...)
    umount /dev/sdb1
    umount /dev/sdb2
    # grabar el pendrive
    sudo dd bs=1M if=loquesea.img of=dev/sdb status=progress
#+END_SRC
